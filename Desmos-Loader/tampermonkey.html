// ==UserScript== // @name Embeddable Desmos // @namespace http://tampermonkey.net/ // @version 0.5 // @description Teachers cannot tell you are cheating when you use this script to embed Desmos on your math quiz:) Search using input and button. Hide popup with right arrow key! // @author H3LLF1R3 // @match https://canvas.instructure.com/courses/* // @grant none // ==/UserScript== function addit(){ document.body.innerHTML += `

AU=Resize; AD=LoadIt; AR=Hide; AL=NewSrc;

` } document.addEventListener('keyup', function(event){ if (event.code == 'ArrowDown'){ addit(); } if (event.code == 'ArrowRight'){ if (document.getElementById('desmos').style.display == 'none'){ document.getElementById('desmos').style.display='block'; }else if (document.getElementById('desmos').style.display == 'block'){ document.getElementById('desmos').style.display='none'; } } if (event.code == 'ArrowUp'){ document.getElementById('desmos').style.width='20%'; document.getElementById('desmos').style.height='30%'; } if (event.code == 'ArrowLeft'){ var newSrc = prompt('Enter a new source (Valid URL)'); document.getElementById('desmos').src=newSrc; } });
